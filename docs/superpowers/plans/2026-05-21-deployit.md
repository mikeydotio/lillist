# `deployit` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Lillist's local `Tools/Deploy/deploy-ios.sh` strategy with a reusable `/deployit` Claude Code plugin published from `mikeydotio/agentics`, backed by a git-based shared build index in a new `mikeydotio/deployit-index` repo, supporting iOS / macOS / visionOS deploys to any tailnet device.

**Architecture:** Per-machine Python http backend (launchd-supervised) serves binaries via Tailscale Serve. After each deploy, the plugin appends a build entry to a local clone of the shared index repo and pushes to GitHub. The listing endpoint `git pull`s before rendering so any Mac's URL shows the union of all tailnet builds. Reference: `docs/plans/2026-05-21-deployit-design.md`.

**Tech Stack:** Bash + Python 3 (stdlib only — no pip), git, launchd, Tailscale Serve, xcodebuild, hdiutil, xcrun notarytool. Test harness: bash, modeled after `mikeydotio/agentics/plugins/semver/tests/`.

```python
#!/usr/bin/env python3
"""deployit CLI — orchestrates archive/export/serve/push.

Subcommands:
  bootstrap          one-time per-machine setup
  deploy             archive and publish a build
  list               show recent builds across the tailnet
  url                print this machine's listing URL
  status             backend health + recent local deploys
  gc                 prune local artifacts; archive in index
"""

import argparse
import json
import sys
from pathlib import Path

__version__ = "0.1.0"


def ok(payload):
    payload["ok"] = True
    print(json.dumps(payload, indent=2))


def fail(message, **extra):
    payload = {"ok": False, "display": message, **extra}
    print(json.dumps(payload, indent=2))
    sys.exit(1)


def not_implemented(name):
    fail(f"{name}: not implemented yet")


def cmd_bootstrap(args):     not_implemented("bootstrap")
def cmd_deploy(args):        not_implemented("deploy")
def cmd_list(args):          not_implemented("list")
def cmd_url(args):           not_implemented("url")
def cmd_status(args):        not_implemented("status")
def cmd_gc(args):            not_implemented("gc")


def build_parser():
    parser = argparse.ArgumentParser(prog="deployit")
    parser.add_argument("--plugin-root", required=True, type=Path)
    parser.add_argument("--version", action="store_true")
    sub = parser.add_subparsers(dest="cmd")

    sub.add_parser("bootstrap").set_defaults(func=cmd_bootstrap)

    p_deploy = sub.add_parser("deploy")
    p_deploy.add_argument("--platform", choices=["ios", "macos", "visionos"])
    p_deploy.add_argument("--scheme")
    p_deploy.set_defaults(func=cmd_deploy)

    p_list = sub.add_parser("list")
    p_list.add_argument("--platform", choices=["ios", "macos", "visionos"])
    p_list.add_argument("--project")
    p_list.set_defaults(func=cmd_list)

    sub.add_parser("url").set_defaults(func=cmd_url)
    sub.add_parser("status").set_defaults(func=cmd_status)

    p_gc = sub.add_parser("gc")
    p_gc.add_argument("--keep", type=int)
    p_gc.add_argument("--older-than", type=int, help="days")
    p_gc.set_defaults(func=cmd_gc)

    return parser


def main(argv):
    args = build_parser().parse_args(argv)
    if args.version:
        ok({"version": __version__})
        return
    if not args.cmd:
        fail("no subcommand given")
    args.func(args)


if __name__ == "__main__":
    main(sys.argv[1:])
```

```python
#!/usr/bin/env python3
"""deployit backend — long-running http server.

Endpoints:
  GET  /deployit/                       listing (git pull + render)
  GET  /deployit/<build_id>/            per-build landing page
  GET  /deployit/<build_id>/<artifact>  manifest.plist / ipa / dmg
  POST /deployit/_internal/refresh      force git pull
  GET  /deployit/_healthz               liveness probe

Stdlib only. Designed to be supervised by launchd.
"""

import argparse
import json
import sys
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer
from pathlib import Path


class Handler(BaseHTTPRequestHandler):
    root: Path = Path()  # set on subclass at startup

    def log_message(self, fmt, *args):
        sys.stderr.write("backend: " + fmt % args + "\n")

    def _send_json(self, status, payload):
        body = json.dumps(payload).encode()
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(body)))
        self.end_headers()
        self.wfile.write(body)

    def do_GET(self):
        if self.path == "/deployit/_healthz":
            self._send_json(200, {"ok": True})
            return
        self._send_json(404, {"ok": False, "error": "not_found"})


def make_handler(root: Path):
    return type("Handler", (Handler,), {"root": root})


def main(argv):
    p = argparse.ArgumentParser()
    p.add_argument("--port", type=int, required=True)
    p.add_argument("--root", type=Path, required=True)
    args = p.parse_args(argv)

    handler_cls = make_handler(args.root)
    server = ThreadingHTTPServer(("127.0.0.1", args.port), handler_cls)
    sys.stderr.write(f"backend: listening on 127.0.0.1:{args.port}, root={args.root}\n")
    server.serve_forever()


if __name__ == "__main__":
    main(sys.argv[1:])
```

```python
import datetime as _dt
import html
import string
import subprocess
import threading
from urllib.parse import urlparse


def _read_index(root: Path):
    try:
        return json.loads((root / "index" / "builds.json").read_text())
    except FileNotFoundError:
        return {"version": 1, "builds": []}


def _relative_time(iso: str) -> str:
    when = _dt.datetime.fromisoformat(iso)
    now = _dt.datetime.now(when.tzinfo)
    delta = now - when
    secs = int(delta.total_seconds())
    if secs < 60:        return f"{secs}s ago"
    if secs < 3600:      return f"{secs // 60}m ago"
    if secs < 86400:     return f"{secs // 3600}h ago"
    return f"{secs // 86400}d ago"


def _install_link(build):
    inst = build["install"]
    if inst["kind"] == "itms-services":
        href = "itms-services://?action=download-manifest&url=" + inst["manifest_url"]
        return href, "Install"
    return inst["dmg_url"], "Download"


_ROW_TEMPLATE = string.Template("""<li class="$archived_class">
  <div class="row">
    <span class="title">$project · $platform · $version (build $build_number)</span>
    <a class="install" href="$install_href">$install_label</a>
  </div>
  <div class="sub">$origin_host · $commit · $rel</div>
</li>""")


_pull_lock = threading.Lock()
_last_pull = "never"


def _git_pull(root: Path):
    """Pull the index repo. Best-effort; logs and returns on failure."""
    global _last_pull
    with _pull_lock:
        try:
            subprocess.run(
                ["git", "-C", str(root / "index"), "pull", "--rebase", "--quiet"],
                check=True, timeout=15,
            )
            _last_pull = _dt.datetime.now().isoformat(timespec="seconds")
        except (subprocess.CalledProcessError, subprocess.TimeoutExpired) as exc:
            sys.stderr.write(f"backend: git pull failed: {exc}\n")


def _render_listing(root: Path, local_host: str) -> bytes:
    data = _read_index(root)
    rows = []
    for b in data["builds"]:
        href, label = _install_link(b)
        rows.append(_ROW_TEMPLATE.substitute(
            archived_class="archived" if b.get("archived") else "",
            project=html.escape(b["project"]),
            platform=html.escape(b["platform"]),
            version=html.escape(b["marketing_version"]),
            build_number=html.escape(b["build_number"]),
            install_href=html.escape(href, quote=True),
            install_label=label,
            origin_host=html.escape(b["origin_host"]),
            commit=html.escape(b["commit"][:7]),
            rel=html.escape(_relative_time(b["timestamp"])),
        ))
    template_path = root / "_plugin_root" / "assets" / "listing.template.html"
    tmpl = string.Template(template_path.read_text())
    return tmpl.substitute(
        LAST_PULL=html.escape(f"last pull: {_last_pull}"),
        TOTAL_BUILDS=str(len(data["builds"])),
        LOCAL_HOST=html.escape(local_host),
        ROWS="\n".join(rows),
    ).encode()
```

```python
import os
import shutil
import subprocess
from textwrap import dedent


def _state_dir() -> Path:
    override = os.environ.get("DEPLOYIT_STATE_DIR")
    if override:
        return Path(override)
    return Path.home() / "Library" / "Application Support" / "deployit"


def _tailscale_bin() -> str:
    return os.environ.get("DEPLOYIT_TAILSCALE_BIN", "tailscale")


def _derive_base_url() -> str:
    """Run `tailscale status --json` and build the listing base URL."""
    res = subprocess.run(
        [_tailscale_bin(), "status", "--json"],
        capture_output=True, text=True, check=True, timeout=5,
    )
    data = json.loads(res.stdout)
    dns = data["Self"]["DNSName"].rstrip(".")
    return f"https://{dns}/deployit"


def _write_config(state: Path, plugin_root: Path, base_url: str) -> Path:
    cfg = state / "config.toml"
    if cfg.exists():
        return cfg
    template = (plugin_root / "assets" / "config.example.toml").read_text()
    template = template.replace(
        'base_url = "https://<host>.<tailnet>.ts.net/deployit"',
        f'base_url = "{base_url}"',
    )
    cfg.write_text(template)
    return cfg


def _ensure_dirs(state: Path):
    for sub in ("serve", "index", "logs", "bin"):
        (state / sub).mkdir(parents=True, exist_ok=True)


def _symlink_plugin_root(state: Path, plugin_root: Path):
    link = state / "_plugin_root"
    if link.is_symlink() or link.exists():
        link.unlink()
    link.symlink_to(plugin_root)


def _clone_index(state: Path):
    if os.environ.get("DEPLOYIT_SKIP_INDEX_CLONE"):
        (state / "index" / "builds.json").write_text(
            json.dumps({"version": 1, "builds": []}))
        return
    index_dir = state / "index"
    if (index_dir / ".git").is_dir():
        return
    # Clean and clone fresh; the placeholder builds.json from _ensure_dirs is fine to overwrite.
    shutil.rmtree(index_dir)
    subprocess.run(
        ["git", "clone", "https://github.com/mikeydotio/deployit-index.git", str(index_dir)],
        check=True, timeout=30,
    )


def _configure_tailscale_serve(port: int):
    if os.environ.get("DEPLOYIT_SKIP_TAILSCALE_SERVE"):
        return
    subprocess.run(
        [_tailscale_bin(), "serve", "--bg", str(port)],
        check=False, timeout=10,
    )


def _launchd_plist(plugin_root: Path, state: Path, port: int) -> str:
    return dedent(f"""\
        <?xml version="1.0" encoding="UTF-8"?>
        <!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
        <plist version="1.0">
        <dict>
          <key>Label</key><string>com.mikeydotio.deployit.backend</string>
          <key>ProgramArguments</key>
          <array>
            <string>/usr/bin/env</string>
            <string>python3</string>
            <string>{plugin_root}/bin/deployit-backend</string>
            <string>--port</string><string>{port}</string>
            <string>--root</string><string>{state}</string>
            <string>--plugin-root</string><string>{plugin_root}</string>
          </array>
          <key>RunAtLoad</key><true/>
          <key>KeepAlive</key><true/>
          <key>StandardOutPath</key><string>{state}/logs/backend.log</string>
          <key>StandardErrorPath</key><string>{state}/logs/backend.err.log</string>
        </dict>
        </plist>
        """)


def _install_launchd(plugin_root: Path, state: Path, port: int):
    if os.environ.get("DEPLOYIT_SKIP_LAUNCHD"):
        return
    plist_dir = Path.home() / "Library" / "LaunchAgents"
    plist_dir.mkdir(parents=True, exist_ok=True)
    plist_path = plist_dir / "com.mikeydotio.deployit.backend.plist"
    plist_path.write_text(_launchd_plist(plugin_root, state, port))
    launchctl = os.environ.get("DEPLOYIT_LAUNCHCTL_BIN", "launchctl")
    subprocess.run([launchctl, "bootout", f"gui/{os.getuid()}/com.mikeydotio.deployit.backend"],
                   check=False)
    subprocess.run([launchctl, "bootstrap", f"gui/{os.getuid()}", str(plist_path)],
                   check=False)


def cmd_bootstrap(args):

    state.mkdir(parents=True, exist_ok=True)

    _install_launchd(args.plugin_root, state, 8729)
    ok({
        "display": f"deployit bootstrapped. Bookmark on iPhone: {base_url}/",
        "state_dir": str(state),
        "config_path": str(cfg_path),
        "base_url": base_url,
    })
```

```python
import re as _re


def _find_workspace(cwd: Path) -> Path | None:
    for p in cwd.glob("*.xcworkspace"):
        return p
    return None


def _find_project_yml(cwd: Path, platform: str) -> Path | None:
    """Return the first project.yml under Apps/ whose name suggests the platform."""
    candidates = sorted(cwd.glob(f"Apps/*-{platform.title()}/project.yml")) \
              or sorted(cwd.glob(f"Apps/*/project.yml"))
    return candidates[0] if candidates else None


def _read_marketing_version(project_yml: Path) -> str:
    text = project_yml.read_text()
    m = _re.search(r'MARKETING_VERSION:\s*"([^"]+)"', text)
    if not m:
        fail(f"Could not parse MARKETING_VERSION from {project_yml}")
    return m.group(1)


def _read_bundle_id(project_yml: Path) -> str:
    text = project_yml.read_text()
    m = _re.search(r'PRODUCT_BUNDLE_IDENTIFIER:\s*"([^"]+)"', text)
    if not m:
        fail(f"Could not parse PRODUCT_BUNDLE_IDENTIFIER from {project_yml}")
    return m.group(1)


def _short_sha(cwd: Path) -> str:
    res = subprocess.run(["git", "-C", str(cwd), "rev-parse", "--short", "HEAD"],
                         capture_output=True, text=True, check=True)
    return res.stdout.strip()


def _derive_metadata(cwd: Path, platform: str) -> dict:
    workspace = _find_workspace(cwd) or fail("no .xcworkspace in cwd")
    project_yml = _find_project_yml(cwd, platform) or fail(
        f"no Apps/*/project.yml found for platform {platform}")
    return {
        "project": workspace.stem,
        "platform": platform,
        "bundle_id": _read_bundle_id(project_yml),
        "marketing_version": _read_marketing_version(project_yml),
        "commit": _short_sha(cwd),
        "workspace": str(workspace),
        "project_yml": str(project_yml),
    }
```

```python
import datetime as _dt
import shutil as _shutil


def _read_config(state: Path) -> dict:
    """Parse the relevant fields out of config.toml without taking a tomllib dep
    on older Pythons. Python 3.11+ has tomllib; macOS 14 ships 3.9 by default but
    Xcode includes 3.11+. Fall back to regex if tomllib is missing."""
    try:
        import tomllib
        return tomllib.loads((state / "config.toml").read_text())
    except ModuleNotFoundError:
        text = (state / "config.toml").read_text()
        port = int(_re.search(r'port\s*=\s*(\d+)', text).group(1))
        base_url = _re.search(r'base_url\s*=\s*"([^"]+)"', text).group(1)
        notarize = bool(_re.search(r'notarize\s*=\s*true', text))
        notary_profile_m = _re.search(r'notary_profile\s*=\s*"([^"]+)"', text)
        return {
            "server": {"port": port, "base_url": base_url},
            "macos":  {"notarize": notarize,
                       "notary_profile": notary_profile_m.group(1) if notary_profile_m else ""},
        }


def _build_id(project: str, platform: str, commit: str) -> str:
    stamp = _dt.datetime.now().strftime("%Y%m%d-%H%M%S")
    return f"{project.lower()}-{platform}-{stamp}-{commit}"


def _xcodebuild_archive(meta: dict, archive_path: Path, scheme: str | None):
    cmd = [
        "xcodebuild",
        "-workspace", meta["workspace"],
        "-scheme", scheme or f"{meta['project']}-{meta['platform'].upper()}"
                              if meta['platform'] == 'ios'
                              else f"{meta['project']}-{meta['platform']}",
        "-configuration", "Debug",
        "-destination", _destination_for(meta["platform"]),
        "-archivePath", str(archive_path),
        "-allowProvisioningUpdates",
        "archive",
    ]
    res = subprocess.run(cmd)
    if res.returncode != 0:
        fail("xcodebuild archive failed (see Xcode output above for the recipe — "
             "Tools/Deploy/README troubleshooting still applies; see references/troubleshooting.md)")


def _destination_for(platform: str) -> str:
    return {
        "ios":      "generic/platform=iOS",
        "visionos": "generic/platform=visionOS",
        "macos":    "generic/platform=macOS",
    }[platform]


def _read_archive_build_number(archive_path: Path, project: str) -> str:
    plist = archive_path / "Products" / "Applications" / f"{project}.app" / "Info.plist"
    res = subprocess.run(
        ["/usr/libexec/PlistBuddy", "-c", "Print :CFBundleVersion", str(plist)],
        capture_output=True, text=True, check=True,
    )
    return res.stdout.strip()


def _xcodebuild_export(archive_path: Path, export_dir: Path, plugin_root: Path, platform: str):
    options = plugin_root / "assets" / f"ExportOptions.{platform}.plist"
    cmd = [
        "xcodebuild", "-exportArchive",
        "-archivePath", str(archive_path),
        "-exportOptionsPlist", str(options),
        "-exportPath", str(export_dir),
        "-allowProvisioningUpdates",
    ]
    res = subprocess.run(cmd)
    if res.returncode != 0:
        fail("xcodebuild -exportArchive failed (see Xcode output above)")


def _stage_ios_or_visionos(state: Path, plugin_root: Path,
                           build_id: str, ipa_src: Path, meta_full: dict):
    dst = state / "serve" / build_id
    if dst.exists():
        _shutil.rmtree(dst)
    dst.mkdir(parents=True)

    artifact_name = ipa_src.name  # e.g. "Lillist.ipa"
    _shutil.move(str(ipa_src), dst / artifact_name)

    ipa_url      = f"{meta_full['origin_base_url']}/{build_id}/{artifact_name}"
    manifest_url = f"{meta_full['origin_base_url']}/{build_id}/manifest.plist"

    manifest_tmpl = string.Template(
        (plugin_root / "assets" / "manifest.template.plist").read_text())
    (dst / "manifest.plist").write_text(manifest_tmpl.substitute(
        IPA_URL=ipa_url,
        BUNDLE_ID=meta_full["bundle_id"],
        BUNDLE_VERSION=meta_full["marketing_version"],
        TITLE=meta_full["project"],
    ))

    (dst / "_meta.json").write_text(json.dumps({
        **{k: meta_full[k] for k in (
            "id", "platform", "project", "bundle_id",
            "marketing_version", "build_number", "commit",
            "timestamp", "origin_host", "origin_base_url",
        )},
        "install": {
            "kind": "itms-services",
            "manifest_url": manifest_url,
            "ipa_url": ipa_url,
        },
        "primary_artifact": artifact_name,
    }, indent=2))

    return artifact_name, ipa_url, manifest_url
```

```python
import urllib.request


def _append_to_index(state: Path, entry: dict, attempts: int = 3):
    index_dir = state / "index"
    builds_path = index_dir / "builds.json"

    for attempt in range(1, attempts + 1):
        subprocess.run(["git", "-C", str(index_dir), "pull", "--rebase", "--quiet"],
                       check=True, timeout=30)
        data = json.loads(builds_path.read_text())
        data["builds"].insert(0, entry)
        builds_path.write_text(json.dumps(data, indent=2) + "\n")

        msg = (f"feat(deploy): {entry['project']} {entry['platform']} "
               f"{entry['marketing_version']} (build {entry['build_number']}) "
               f"from {entry['origin_host']}")
        subprocess.run(["git", "-C", str(index_dir), "add", "builds.json"], check=True)
        subprocess.run(["git", "-C", str(index_dir), "commit", "-m", msg], check=True)
        push = subprocess.run(
            ["git", "-c", "url.https://github.com/.insteadOf=git@github.com:",
             "-C", str(index_dir), "push", "origin", "main"],
            capture_output=True, text=True,
        )
        if push.returncode == 0:
            return
        # Conflict / rejected — undo local commit and retry from pull.
        subprocess.run(["git", "-C", str(index_dir), "reset", "--hard", "origin/main"],
                       check=True)
        if attempt == attempts:
            fail("push to deployit-index failed after retries; local artifacts retained",
                 stderr=push.stderr.strip())


def _refresh_local_backend(port: int):

        urllib.request.urlopen(req, timeout=3).read()
    except Exception as exc:
        sys.stderr.write(f"warning: local backend refresh failed: {exc}\n")
```

```python
def _stage_macos(state: Path, plugin_root: Path, build_id: str,
                 app_src: Path, meta_full: dict, notarize: bool,
                 notary_profile: str):
    dst = state / "serve" / build_id
    if dst.exists():
        _shutil.rmtree(dst)
    dst.mkdir(parents=True)

    dmg_name = f"{meta_full['project']}.dmg"
    dmg_path = dst / dmg_name
    _shutil.move(str(app_src), dst / app_src.name)  # stage the .app inside dst

    # hdiutil create -srcfolder <.app> -volname <project> -format UDZO <dmg>
    subprocess.run([
        "hdiutil", "create",
        "-srcfolder", str(dst / app_src.name),
        "-volname", meta_full["project"],
        "-format", "UDZO",
        "-ov",
        str(dmg_path),
    ], check=True)
    _shutil.rmtree(dst / app_src.name)

    if notarize:
        if not notary_profile:
            fail("macos.notarize=true but macos.notary_profile is empty in config.toml")
        subprocess.run([
            "xcrun", "notarytool", "submit", str(dmg_path),
            "--keychain-profile", notary_profile,
            "--wait",
        ], check=True)
        subprocess.run(["xcrun", "stapler", "staple", str(dmg_path)], check=True)

    dmg_url = f"{meta_full['origin_base_url']}/{build_id}/{dmg_name}"
    (dst / "_meta.json").write_text(json.dumps({
        **{k: meta_full[k] for k in (
            "id", "platform", "project", "bundle_id",
            "marketing_version", "build_number", "commit",
            "timestamp", "origin_host", "origin_base_url",
        )},
        "install": {"kind": "direct-download", "dmg_url": dmg_url},
        "primary_artifact": dmg_name,
    }, indent=2))
    return dmg_name, dmg_url
```

```python
def cmd_gc(args):
    state = _state_dir()
    cfg = _read_config(state)
    base_url = cfg["server"]["base_url"].rstrip("/")
    builds_path = state / "index" / "builds.json"
    data = json.loads(builds_path.read_text())

    local = [b for b in data["builds"] if b["origin_base_url"] == base_url and not b.get("archived")]
    # Newest first; the index is already newest-first, but resort to be safe.
    local.sort(key=lambda b: b["timestamp"], reverse=True)

    to_archive = []
    if args.keep is not None:
        to_archive = local[args.keep:]
    elif getattr(args, "older_than", None) is not None:
        cutoff = _dt.datetime.now(_dt.timezone.utc) - _dt.timedelta(days=args.older_than)
        to_archive = [b for b in local
                      if _dt.datetime.fromisoformat(b["timestamp"]).astimezone(_dt.timezone.utc) < cutoff]
    else:
        fail("gc requires --keep N or --older-than DAYS")

    removed = 0
    for b in to_archive:
        d = state / "serve" / b["id"]
        if d.exists():
            _shutil.rmtree(d)
            removed += 1
        b["archived"] = True

    builds_path.write_text(json.dumps(data, indent=2) + "\n")
    subprocess.run(["git", "-C", str(state / "index"), "add", "builds.json"], check=True)
    if to_archive:
        subprocess.run(["git", "-C", str(state / "index"), "commit", "-m",
                        f"chore(deploy): archive {len(to_archive)} build(s) from {base_url}"], check=True)
        subprocess.run(["git", "-c", "url.https://github.com/.insteadOf=git@github.com:",
                        "-C", str(state / "index"), "push", "origin", "main"], check=True)
        _refresh_local_backend(cfg["server"]["port"])

    ok({"display": f"archived {len(to_archive)} build(s); removed {removed} local dir(s)"})
```

```markdown
# deployit Orchestrator

`/deployit` archives the current Xcode workspace, exports a
Development- or Developer-ID-signed binary, stages it under
`~/Library/Application Support/deployit/serve/`, appends an entry to
the shared `mikeydotio/deployit-index` repo, and points the user at a
Tailscale-served URL.

**Router:** `bash ${CLAUDE_PLUGIN_ROOT}/bin/deployit-router.sh <ARGUMENTS>`

## Hard rules

1. Route every command through the router. Never call `deployit-cli` directly.
2. If the CLI returns `ok: false`, show the `display` (or `message`) and stop.
3. Never push to the index repo from this orchestrator; that is the CLI's job.
4. Build-number bumping is the host repo's responsibility. Lillist's
   `Tools/Deploy/bump-build-number.sh` Archive pre-action stays; the plugin
   reads the resolved `CFBundleVersion` from the built `Info.plist`.

## Generic flow

1. **Route**: `bash ${CLAUDE_PLUGIN_ROOT}/bin/deployit-router.sh <ARGUMENTS>`
2. **Check `ok`**: if false, show `display` and stop.
3. **Display**: show the CLI's `display` field on success.

## Per-command notes

- **bootstrap** is idempotent. Run it once on each Mac, and re-run any time
  the plugin version changes (it rewrites the launchd plist with the
  current plugin path).
- **deploy** auto-detects project + bundle ID + marketing version from
  `*.xcworkspace` + `Apps/*/project.yml`. If multiple schemes match the
  platform the CLI surfaces a `questions` array; present it via
  `AskUserQuestion` and re-invoke with `--scheme <chosen>`.
- **list / status / url** are read-only; no confirmations needed.
- **gc** requires `--keep N` or `--older-than D` — never run unqualified.
```
